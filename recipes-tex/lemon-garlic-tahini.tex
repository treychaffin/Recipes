\begin{recipe}{\textbf{Lemon Garlic Tahini Sauce}}{4 servings}{5 minutes}

  \freeform{}\textit{Lemon Garlic Tahini Sauce}

  \ing[\fr14]{cup}{Tahini}
  \ing[2]{cloves}{Garlic}
  \ing[3]{tbsp}{Extra Virgin Olive Oil}
  \ing[3]{tbsp}{Water}
  \ing[1 \fr12]{tbsp}{Lemon Juice}
  \ing[\fr14]{tsp}{Cumin}
  \ing[\fr18]{tsp}{Smoked Paprika}
  \ing[\fr14]{tsp}{Salt}
  \ing[\fr18]{tsp}{Black Pepper}

  Puree the mixture until smooth to form the Lemon Garlic Tahini Sauce
