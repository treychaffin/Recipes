\begin{recipe}{\textbf{Chili}}{8 servings}{15m prep time, 45m cook time}

  \freeform{}\textit{Chili}

  \ing[4]{tbsp}{Olive Oil}
  \ing[2]{cloves}{Garlic, finely chopped}
  \ing[4]{tbsp}{Shallots, chopped}
  \ing[4]{stalks}{Celery, chopped}

  Briefly saut\'e garlic, shallots, and celery in olive oil using a stockpot

  \ing[1]{pound}{Ground Beef}
  \ing[\fr14]{pound}{Pork Sausage}

  Add in the ground beef and pork sausage, cook until browned.

  \ing[2]{}{Green Bell Peppers, chopped}

  Add in the bell peppers, continue cooking until the peppers are soft

  \ing[4]{tbsp}{Better Than Bouillon Chili Base}
  \ing[\fr12]{cup}{Red Wine}
  \ing[\fr12]{cup}{Kidney Beans}
  \ing[2]{tbsp}{Chili Powder, or to taste}
  \ing[]{}{Salt and Pepper, to taste}

  Add the remaining ingredients, simmer for 30 mins

  \ing[4]{ounces}{Cheddar Cheese, grated}

  Serve sprinkled with cheddar cheese

  \freeform{}\hrulefill{}

\end{recipe}
