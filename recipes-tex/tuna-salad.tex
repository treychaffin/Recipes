\begin{recipe}{\textbf{Tuna Salad Sandwich}}{2 sandwiches}{15m prep time, 0min cook time}
  
  \freeform{}\textit{Tuna Salad Sandwich}

  \ing[5]{ounce}{Tuna}
  \ing[\fr12]{cup}{Mayonnaise}
  \ing[1]{tbsp}{Dijon Mustard}
  \ing[1]{stalk}{Celery, chopped}
  \ing[\fr14]{medium}{Onion}
  \ing[\fr14]{}{Pickles, chopped}
  \ing[]{}{Celery Seed}
  \ing[]{}{Black Pepper}
  \ing[]{}{Paprika}
  \ing[]{}{Salt}

  Mix the ingredients together in a bowl.

  \ing[4]{slices}{Bread, sliced toasted}
  \ing[]{}{Arugula}
  \ing[]{}{Tomato, sliced}

  Serve the tuna salad mixture on toasted bread with arugula and a sliced tomato.

  
  \freeform{}\hrulefill{}

\end{recipe}
