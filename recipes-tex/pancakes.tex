  \freeform{}\textit{Cream Cheese \& Almond Flour Pancakes}

  \ing[2]{tbsp}{\href{https://www.dillons.com/p/kroger-original-cream-cheese/0001111089202}
  {cream cheese}}

  {egg}}

  {almond flour}}
  \ing[2]{scoops}{\href{https://www.amazon.com/gp/product/B07PMLB7SP/}
  {monk fruit}}
  \ing[2]{tbsp}{\href{https://www.dillons.com/p/kerrygold-unsalted-pure-irish-butter/0076770700125}
  {butter}}
