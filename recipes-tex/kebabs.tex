\begin{recipe}{\textbf{Kefta Kebabs}}{}{}

  \freeform{}\textit{Moroccan Kebabs}

  \ing[1]{pound}{Ground Beef, Ground Lamb, or Combination}
  \ing[3]{ounces}{Beef/Lamb Fat, optional}
  \ing[1]{medium}{Onion, chopped very fine or grated}
  \ing[2]{tsp}{Paprika}
  \ing[1]{tsp}{Salt}
  \ing[\fr14]{tsp}{Black Pepper}
  \ing[\fr18]{tsp}{Cayenne Pepper}
  \ing[\fr14]{cup}{Fresh Parsley, chopped}
  \ing[\fr14]{cup}{Fresh Cilantro, chopped}
  \ing[1]{tsp}{Ground Cinnamon}
  \ing[1]{tbsp}{Fresh Mint Leaves, chopped}
  \ing[]{}{Bamboo Skewers, soaked}

  Mix all the ingredients together. Chill the mix for an hour or longer.
  Shape kefta into cylinders, skewer the meat. Grill the skewers.

  \freeform{}\hrulefill{}

\end{recipe}
